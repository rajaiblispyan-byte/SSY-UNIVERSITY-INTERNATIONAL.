function updateStudentList(){
    const emptyEl=document.getElementById('emptyState');
    const cardsEl=document.getElementById('studentCards');

    // Stats
    document.getElementById('totalPendaftar').textContent=registeredStudents.length;
    document.getElementById('totalPhysics').textContent=registeredStudents.filter(s=>s.fakultas.includes('Physics')).length;
    document.getElementById('totalBio').textContent=registeredStudents.filter(s=>s.fakultas.includes('Biosurfaces')).length;
    document.getElementById('totalAero').textContent=registeredStudents.filter(s=>s.fakultas.includes('Aerospace')).length;

    if(registeredStudents.length===0){
        emptyEl.classList.remove('hidden');
        cardsEl.classList.add('hidden');
        return;
    }

    emptyEl.classList.add('hidden');
    cardsEl.classList.remove('hidden');

    const gameEmojis={'Roblox':'🧱','Mobile Legends':'⚔️','PUBG':'🎒','Free Fire':'🔥','Valorant':'🎯','Minecraft':'⛏️'};

    let html='';
    registeredStudents.forEach((s,i)=>{
        const facInfo = s.fakultas.includes('Physics')
            ? {short:'Physics & Quantum', color:'bg-indigo-100 text-primary', icon:'lucide:atom', iconBg:'bg-indigo-50', iconColor:'text-primary'}
            : s.fakultas.includes('Biosurfaces')
            ? {short:'Biosurfaces & Nano', color:'bg-purple-100 text-accent', icon:'lucide:microscope', iconBg:'bg-purple-50', iconColor:'text-accent'}
            : {short:'Aerospace & Astro', color:'bg-pink-100 text-pink-600', icon:'lucide:rocket', iconBg:'bg-pink-50', iconColor:'text-pink-500'};

        const gameBadges = s.games.length > 0
            ? s.games.map(g => `<span class="inline-flex items-center gap-1 bg-slate-100 text-slate-600 text-[10px] font-semibold px-2 py-0.5 rounded-full">${gameEmojis[g]||'🎮'} ${g}</span>`).join('')
            : '<span class="text-[11px] text-slate-300 italic">Tidak ada</span>';

        html += `
        <div class="student-entry slide-up bg-white rounded-2xl border border-slate-200 card-shadow overflow-hidden" style="animation-delay:${i*0.08}s">
            <div class="p-5 md:p-6">
                <!-- Top row: number + photo + name + ID -->
                <div class="flex items-center gap-4 mb-4">
                    <div class="text-xs font-bold text-slate-300 w-6 text-center flex-shrink-0">${i+1}</div>
                    <div class="w-12 h-12 rounded-xl flex-shrink-0 overflow-hidden flex items-center justify-center ${s.photo?'':'bg-indigo-50 border border-indigo-100'}">
                        ${s.photo
                            ? `<img src="${s.photo}" class="w-full h-full object-cover" alt="${s.nama}">`
                            : `<iconify-icon icon="lucide:user" width="20" class="text-slate-300"></iconify-icon>`
                        }
                    </div>
                    <div class="flex-1 min-w-0">
                        <div class="text-base font-bold text-slate-800 truncate">${s.nama}</div>
                        <div class="text-[11px] text-slate-400">${s.gender} • ${s.umur} tahun</div>
                    </div>
                    <div class="text-right flex-shrink-0 hidden sm:block">
                        <div class="text-[9px] font-semibold tracking-[0.15em] uppercase text-slate-300">Student ID</div>
                        <div class="text-sm font-bold font-mono text-slate-700 tracking-wider">${s.nim}</div>
                    </div>
                </div>

                <!-- Detail row: badges -->
                <div class="flex items-center gap-2 mb-3 pl-10">
                    <span class="inline-flex items-center gap-1.5 ${facInfo.color} text-[11px] font-semibold px-3 py-1.5 rounded-lg">
                        <iconify-icon icon="${facInfo.icon}" width="12"></iconify-icon>
                        ${facInfo.short}
                    </span>
                    <span class="inline-flex items-center gap-1.5 bg-emerald-50 text-emerald-700 text-[11px] font-semibold px-3 py-1.5 rounded-lg">
                        <iconify-icon icon="lucide:map-pin" width="12"></iconify-icon>
                        ${s.daerah}
                    </span>
                </div>

                <!-- Games row -->
                <div class="flex items-center gap-2 flex-wrap pl-10">

                <!-- Mobile Student ID -->
